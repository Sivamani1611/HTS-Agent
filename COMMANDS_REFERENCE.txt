================================================================================
                            QUICK START COMMANDS
================================================================================

# RECOMMENDED: Simple Launcher (Interactive Menu)
python launch_hts.py

# Quick Status Check
python launch_hts.py --status

# Quick Help
python launch_hts.py --help

================================================================================
                            WEB APPLICATIONS
================================================================================

# Method 1: Using Simple Launcher (RECOMMENDED)
python launch_hts.py --web ultimate      # Launch Ultimate Web App (full features)
python launch_hts.py --web working       # Launch Working Web App (stable backup)

# Method 2: Direct Streamlit Launch
streamlit run app_ultimate.py            # Ultimate Web App (70KB, 1876 lines)
streamlit run app_working.py             # Working Web App (24KB, 558 lines)

# Method 3: Using Project Manager
python manage_project.py --run-app ultimate
python manage_project.py --run-app working

================================================================================
                            CLI APPLICATIONS
================================================================================

# Method 1: Using Simple Launcher (RECOMMENDED)
python launch_hts.py --cli enhanced      # Enhanced CLI with advanced features
python launch_hts.py --cli interactive   # Interactive CLI with chat interface

# Method 2: Direct CLI Launch
python main_fixed.py --chat              # Enhanced CLI - Interactive mode
python cli_simple.py                     # Interactive CLI - Chat interface

# Method 3: Using Project Manager
python manage_project.py --run-cli enhanced
python manage_project.py --run-cli interactive

================================================================================
                        ENHANCED CLI FEATURES
================================================================================

# Basic Operations
python main_fixed.py --help              # Show all available options
python main_fixed.py --list              # List all HTS codes
python main_fixed.py --stats             # Show database statistics

# Questions & Queries
python main_fixed.py --query "What is GSP?"                    # Ask a question
python main_fixed.py --chat                                    # Interactive chat mode

# Duty Calculations
python main_fixed.py --calc 0101.30.00.00 10000               # Basic calculation
python main_fixed.py --calc 0101.30.00.00 10000 500 100       # With freight & insurance

# Advanced Search & Analysis
python main_fixed.py --search "cattle"                         # Search HTS codes
python main_fixed.py --search "beef"                           # Search by keyword
python main_fixed.py --compare 0101.30.00.00 0102.21.00.00    # Compare duty rates

# Batch Processing
python main_fixed.py --template                                # Generate CSV template
python main_fixed.py --batch mydata.csv                        # Process batch file
python main_fixed.py --batch hts_batch_template.csv            # Process template

# Data Export
python main_fixed.py --export json                             # Export as JSON
python main_fixed.py --export csv                              # Export as CSV

================================================================================
                        INTERACTIVE CLI COMMANDS
================================================================================

# Start Interactive Mode
python main_fixed.py --chat
python cli_simple.py

# Commands Available in Interactive Mode:
help                                      # Show available commands
list                                      # List HTS codes
stats                                     # Show statistics
search cattle                             # Search for products
compare 0101.30.00.00 0102.21.00.00      # Compare rates
calc 0101.30.00.00 10000 500 100         # Calculate duty
template                                  # Generate CSV template
batch myfile.csv                         # Process batch calculations
export json                              # Export database as JSON
export csv                               # Export database as CSV
exit                                      # Exit interactive mode
